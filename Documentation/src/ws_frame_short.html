class interface
	WS_FRAME

create 
	make,
	make_text,
	make_binary,

feature -- Access

	opcode: INTEGER_32
			-- Frame opcode (0x0-0xF).

	payload: ARRAY [NATURAL_8]
			-- Frame payload data.

	is_fin: BOOLEAN
			-- Is this the final fragment?

	is_masked: BOOLEAN
			-- Is payload masked?

	mask_key: ARRAY [NATURAL_8]
			-- 4-byte masking key.
	
feature -- Derived Access

	payload_length: INTEGER_32
			-- Length of payload.
		ensure
			non_negative: Result >= 0

	text_payload: STRING_8
			-- Payload as text (for text frames).
		require
			is_text: opcode = Opcode_text or opcode = Opcode_continuation

	close_code: INTEGER_32
			-- Close status code (for close frames).
		require
			is_close: opcode = Opcode_close
			has_code: payload.count >= 2

	close_reason: STRING_8
			-- Close reason (for close frames).
		require
			is_close: opcode = Opcode_close
	
feature -- Status

	is_control: BOOLEAN
			-- Is this a control frame?

	is_text: BOOLEAN
			-- Is this a text data frame?

	is_binary: BOOLEAN
			-- Is this a binary data frame?

	is_close: BOOLEAN
			-- Is this a close frame?

	is_ping: BOOLEAN
			-- Is this a ping frame?

	is_pong: BOOLEAN
			-- Is this a pong frame?

	is_continuation: BOOLEAN
			-- Is this a continuation frame?
	
feature -- Modification

	set_mask (a_key: ARRAY [NATURAL_8])
			-- Set masking key and enable masking.
		require
			key_not_void: a_key /= Void
			key_length: a_key.count = 4
		ensure
			masked: is_masked
			key_set: mask_key = a_key

	to_bytes: ARRAY [NATURAL_8]
			-- Encode frame to wire format.

	is_valid_opcode (a_opcode: INTEGER_32): BOOLEAN
			-- Is opcode valid per RFC 6455?
	
feature -- Opcodes (RFC 6455 Section 5.2)

	Opcode_continuation: INTEGER_32 = 0

	Opcode_text: INTEGER_32 = 1

	Opcode_binary: INTEGER_32 = 2

	Opcode_close: INTEGER_32 = 8

	Opcode_ping: INTEGER_32 = 9

	Opcode_pong: INTEGER_32 = 10
	
feature -- Close Codes (RFC 6455 Section 7.4.1)

	Close_normal: INTEGER_32 = 1000

	Close_going_away: INTEGER_32 = 1001

	Close_protocol_error: INTEGER_32 = 1002

	Close_unsupported_data: INTEGER_32 = 1003

	Close_invalid_payload: INTEGER_32 = 1007

	Close_policy_violation: INTEGER_32 = 1008

	Close_message_too_big: INTEGER_32 = 1009

	Close_server_error: INTEGER_32 = 1011

	mask_key_not_void: mask_key /= Void
	mask_key_length: mask_key.count = 4
